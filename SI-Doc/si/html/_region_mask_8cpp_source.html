279 {

282  unsigned int polysize = contour.size();
283 
284  int gap = 1;
285  double y;
286 
287  min.x = d_tlc_aabb_x;
288  min.y = d_tlc_aabb_y;
289

294  {
295  if (p.x < min.x)

297 
298  if (p.y < min.y)
299  min.y = p.y;
300 
301  if (p.x > max.x)
302  max.x = p.x;
303 
304  if (p.y > max.y)
305  max.y = p.y;
306  }
307 
308  // Bounds check
309  if ((max.x < 0) || (min.x > d_canvas_width) || (max.y < 0) || (min.y > d_canvas_height))
310  return;
311 
312  // Vertically clip
313  if (min.y < d_tlc_aabb_y)
314  min.y = d_tlc_aabb_y;
315 
316  if (max.y > d_brc_aabb_y)
317  max.y = d_brc_aabb_y;
318 
319  // start on the outside of the object we step out by 1.
320  min.x -= 1;
321  max.x += 1;
322 
323  // Initialise starting conditions
324  y = min.y;
325 
326  // Go through each scan line iteratively, jumping by 'gap' pixels each time
327  while (y < max.y)
328  {
329  scan_hits.clear();
330 
331  int jump = 1;
332  glm::vec2 fp(contour[0].x, contour[0].y);
333 
334  for (size_t i = 0; i < polysize - 1; i++)
335  {
336  glm::vec2 pa(contour[i].x, contour[i].y);
337  glm::vec2 pb(contour[i + 1].x, contour[i + 1].y);
338 
339  // jump double/dud points
340  if (pa.x == pb.x && pa.y == pb.y)
341  continue;
342 
343  // if we encounter our hull/poly start point, then we've now created the
344  // closed
345  // hull, jump the next segment and reset the first-point
346  if ((!jump) && (fp.x == pb.x) && (fp.y == pb.y))
347  {
348  if (i < polysize - 2)
349  {
350  fp.x = contour[i + 2].x;
351  fp.y = contour[i + 2].y;
352  jump = 1;
353  i++;
354  }
355  }
356  else
357  jump = 0;
358 
359  // test to see if this segment makes the scan-cut.
360  if ((pa.y > pb.y && y < pa.y && y > pb.y) || (pa.y < pb.y && y > pa.y && y < pb.y))
361  {

372  }
373  }
