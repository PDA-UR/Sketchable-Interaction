287 void RegionMask::scanlinefill(const std::vector<glm::vec3>& contour, const std::vector<glm::vec3>& aabb)
288 {
289  std::vector<glm::vec2> scan_hits;
290  static glm::vec2 min, max;
291  unsigned int polysize = contour.size();
292 
293  min.x = d_tlc_aabb_x;
294  min.y = d_tlc_aabb_y;
295 
296  max.x = d_brc_aabb_x;
297  max.y = d_brc_aabb_y;
298 
299  // Bounds check
300  if ((max.x < 0) || (min.x > d_canvas_width) || (max.y < 0) || (min.y > d_canvas_height))
301  return;
302 
303  // Vertically clip
304  if (min.y < d_tlc_aabb_y)
305  min.y = d_tlc_aabb_y;
306 
307  if (max.y > d_brc_aabb_y)
308  max.y = d_brc_aabb_y;
309 
310  // start on the outside of the object we step out by 1.
311  min.x -= 1;
312  max.x += 1;
313 
314  // Go through each scan line iteratively, jumping by 'gap' pixels each time
315  for(int32_t y = min.y; y < max.y; ++y)
316  {
317  scan_hits.clear();
318 
319  bool jump = true;
320  glm::vec2 fp(contour[0].x, contour[0].y);
321 
322  for (size_t i = 0; i < polysize - 1; i++)
323  {
324  glm::vec2 pa(contour[i].x, contour[i].y);
325  glm::vec2 pb(contour[i + 1].x, contour[i + 1].y);
326 
327  // jump double/dud points
328  if (pa.x == pb.x && pa.y == pb.y)
329  continue;
330 
331  // if we encounter our hull/poly start point, then we've now created the
332  // closed
333  // hull, jump the next segment and reset the first-point
334  if ((!jump) && (fp.x == pb.x) && (fp.y == pb.y))
335  {
336  if (i < polysize - 2)
337  {
338  fp.x = contour[i + 2].x;
339  fp.y = contour[i + 2].y;
340  jump = true;
341  i++;
342  }
343  }
344  else
345  jump = false;
346 
347  // test to see if this segment makes the scan-cut.
348  if ((pa.y > pb.y && y < pa.y && y > pb.y) || (pa.y < pb.y && y > pa.y && y < pb.y))
349  scan_hits.emplace_back(pa.x == pb.x ? pa.x : (pb.x - pa.x) / (pb.y - pa.y) * (y - pa.y) + pa.x, y);
350  }
351 
352  // Sort the scan hits by X, so we have a proper left->right ordering
353  std::sort(scan_hits.begin(), scan_hits.end(), [&](glm::vec2 const &a, glm::vec2 const &b)
354  {
355  return a.x < b.x;
356  });
357 
358  int32_t l = scan_hits.size() - 1;
359 
360  // generate the line segments.
361  for (int32_t i = 0; i < l; i += 2)
362  for(int32_t x = scan_hits[i].x; x < scan_hits[i + 1].x; x++)
363  this->set_bit(glm::vec3(x, y, 1));
364  }
365 
366  scan_hits.clear();
367 }
